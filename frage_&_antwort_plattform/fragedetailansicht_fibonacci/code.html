Ja, du kannst Memoization verwenden, um die Leistung drastisch zu verbessern. Die Idee ist, die Ergebnisse der Fibonacci-Zahlen zu speichern, sobald sie berechnet wurden, damit du sie nicht erneut berechnen musst. 

private Map<Integer, Integer> memo = new HashMap<>();
public int fibonacci(int n) {
    if (n <= 1) {
        return n;
    }
    if (memo.containsKey(n)) {
        return memo.get(n);
    }
    int result = fibonacci(n - 1) + fibonacci(n - 2);
    memo.put(n, result);
    return result;
}
